```function calculaSalario(qtdeCarrosVendidos, valorTotalVendas) {
  let soma = qtdeCarrosVendidos * 100
  let resultado = valorTotalVendas * 0.05
  let total = 2000 + soma + resultado
  
  return total
  

}```
